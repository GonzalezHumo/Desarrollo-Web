Código JavaScript
// Declaración de variables
let a = 18;
let b = 4;

// Operaciones aritméticas
let suma = a + b;
let resta = a - b;
let division = a / b;
let exponenciacion = a ** b;

// Mostrar resultados con innerHTML
document.getElementById("resultado-html").innerHTML =
  "<b>Suma:</b> a + b = " + suma + "<br>" +
  "<b>Resta:</b> a - b = " + resta + "<br>" +
  "<b>División:</b> a / b = " + division + "<br>" +
  "<b>Exponenciación:</b> a ** b = " + exponenciacion;
